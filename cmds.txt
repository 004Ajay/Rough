function abbrp
    set name $argv[1]
    set value $argv[2..-1]
    set line "abbr --add $name '$value'"
    if not grep -q "$line" ~/.config/fish/abbrs.fish
        echo $line >> ~/.config/fish/abbrs.fish
        source ~/.config/fish/abbrs.fish
        echo "Saved: $line"
    else
        echo "Already exists: $line"
    end
end
